<p class="section-label">Why naml?</p>
<p class="section-heading">Built for speed and clarity</p>

<div class="features-grid">
  <div class="feature-card">
    <h3>Cranelift JIT</h3>
    <p>JIT compilation with Cranelift for fast development cycles. Native binary output with LTO for production deployments.</p>
  </div>
  <div class="feature-card">
    <h3>Cross-Platform</h3>
    <p>One codebase, three targets. Compile to native binaries, server-side WASM, or browser WASM.</p>
  </div>
  <div class="feature-card">
    <h3>Type Safe</h3>
    <p>Required type annotations make code self-documenting. No guessing, no inference surprises, no runtime type errors.</p>
  </div>
  <div class="feature-card">
    <h3>Rich Standard Library</h3>

<p class="section-label">See it in action</p>
<p class="section-heading">Concurrency made simple</p>

```naml title="main.nm"
use std::threads::*;

struct Task {
    id: int,
    name: string
}

pub fn (self: Task) run(out: channel<string>) {
    var result: string = fmt("done: {} (id={})", self.name, self.id);
    send(out, result);
}

fn main() {
    var ch: channel<string> = open_channel(10);
    var tasks: [Task] = [
        Task { id: 1, name: "compile" },
        Task { id: 2, name: "optimize" },
        Task { id: 3, name: "deploy" }
    ];

    for (i: int, task: Task in tasks) {
        spawn { task.run(ch); };
    }
    join();

    for (i: int, _: int in [0, 1, 2]) {
        println(receive(ch) ?? "error");
    }
    close(ch);
}
```
