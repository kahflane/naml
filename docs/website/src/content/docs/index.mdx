## Quick Example

```naml
use std::threads::*;

struct Person {
    name: string,
    age: int
}

pub fn (self: Person) greet() -> string {
    return fmt("Hello, I am {} (age {})", self.name, self.age);
}

fn main() {
    var alice: Person = Person { name: "Alice", age: 30 };
    println(alice.greet());

    var ch: channel<string> = open_channel(1);
    spawn {
        send(ch, "world");
    };
    join();

    var msg: string = receive(ch) ?? "nobody";
    println("Hello, {}!", msg);
    close(ch);
}
```
